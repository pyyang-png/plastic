        // --- State Variables ---

        let currentUser = null;

        let userName = "";

        let quizMode = 0;

        let currentQuestions = [];

        let currentIndex = 0;

        let userScore = 0;

        let showExplanation = false;

        let selectedOption = null;

 

        // Initialize Icons

        lucide.createIcons();

 

        // Initialize Authentication

        const initAuth = async () => {

            try {

                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                    await signInWithCustomToken(auth, __initial_auth_token);

                } else {

                    await signInAnonymously(auth);

                }

            } catch (err) {

                console.error("Auth error:", err);

                // Even if auth fails due to missing config locally, allow user to test UI

                switchView('login-view');

            }

        };

 

        initAuth();

 

        onAuthStateChanged(auth, (user) => {

            currentUser = user;

            switchView('login-view');

            if (user) {

                setupLeaderboardListener();

            }

        });

 

        // Setup Realtime Leaderboard Listener

        let leaderboardData = [];

        function setupLeaderboardListener() {

            if (!currentUser) return;

            try {

                const lbRef = collection(db, 'artifacts', appId, 'public', 'data', 'leaderboard');

                onSnapshot(lbRef, (snapshot) => {

                    const data = snapshot.docs.map(doc => ({id: doc.id, ...doc.data()}));

                    // Sort in memory to avoid index requirements

                    data.sort((a, b) => b.score - a.score);

                    leaderboardData = data.slice(0, 10);

                    renderLeaderboard();

                }, (error) => {

                    console.error("Snapshot error:", error);

                });

            } catch(e) {

                console.warn("Firestore not configured properly, leaderboard disabled.");

            }

        }

 

        // --- Functions ---

        window.switchView = function(viewId) {

            document.querySelectorAll('#app > div').forEach(el => el.classList.add('hidden'));

            document.getElementById(viewId).classList.remove('hidden');

        };

 

        window.login = function() {

            const input = document.getElementById('username-input').value.trim();

            if (input) {

                userName = input;

                document.getElementById('display-name').textContent = userName;

                switchView('menu-view');

            }

        };

 

        window.startQuiz = function(mode) {

            quizMode = mode;

            let filtered = [];

            if (mode === 1) {

                filtered = QUESTIONS_DB.filter(q => q.chapter === 1);

                document.getElementById('quiz-mode-tag').textContent = "第一章 化性";

            } else if (mode === 2) {

                filtered = QUESTIONS_DB.filter(q => q.chapter === 2);

                document.getElementById('quiz-mode-tag').textContent = "第二章 物性";

            } else {

                filtered = [...QUESTIONS_DB].sort(() => 0.5 - Math.random()).slice(0, 20);

                document.getElementById('quiz-mode-tag').textContent = "隨機挑戰";

            }

 

            currentQuestions = filtered;

            currentIndex = 0;

            userScore = 0;

            showExplanation = false;

            selectedOption = null;

 

            document.getElementById('current-score').textContent = "0";

            switchView('quiz-view');

            renderQuestion();

        };

 

        window.renderQuestion = function() {

            document.getElementById('quiz-progress').textContent = `${currentIndex + 1} / ${currentQuestions.length}`;

            document.getElementById('progress-bar').style.width = `${((currentIndex + 1) / currentQuestions.length) * 100}%`;

           

            const q = currentQuestions[currentIndex];

            document.getElementById('question-text').textContent = q.q;

           

            const optionsContainer = document.getElementById('options-container');

            optionsContainer.innerHTML = '';

           

            q.options.forEach((opt, idx) => {

                const btn = document.createElement('button');

                btn.className = "w-full text-left p-5 rounded-2xl border-2 border-slate-200 hover:border-blue-400 bg-white transition-all flex items-center gap-4 text-slate-700 option-btn";

                btn.onclick = () => handleOptionSelect(idx, btn);

                btn.innerHTML = `

                    <span class="w-10 h-10 rounded-xl flex items-center justify-center font-black shrink-0 transition-colors bg-slate-100 text-slate-500 option-letter">

                        ${String.fromCharCode(65 + idx)}

                    </span>

                    <span class="text-lg font-medium option-text">${opt}</span>

                    <i data-lucide="check-circle-2" class="ml-auto text-green-500 shrink-0 hidden check-icon w-6 h-6"></i>

                    <i data-lucide="x-circle" class="ml-auto text-red-400 shrink-0 hidden cross-icon w-6 h-6"></i>

                `;

                optionsContainer.appendChild(btn);

            });

 

            document.getElementById('explanation-box').classList.add('hidden');

            lucide.createIcons();

        };

 

        window.handleOptionSelect = function(idx, clickedBtn) {

            if (showExplanation) return;

            selectedOption = idx;

            showExplanation = true;

           

            const q = currentQuestions[currentIndex];

            const isCorrect = (idx === q.ans);

           

            if (isCorrect) {

                userScore++;

                document.getElementById('current-score').textContent = userScore;

            }

 

            // Style all buttons

            const buttons = document.querySelectorAll('.option-btn');

            buttons.forEach((btn, buttonIdx) => {

                btn.disabled = true;

                const letter = btn.querySelector('.option-letter');

                const text = btn.querySelector('.option-text');

                const checkIcon = btn.querySelector('.check-icon');

                const crossIcon = btn.querySelector('.cross-icon');

 

                if (buttonIdx === q.ans) {

                    btn.className = "w-full text-left p-5 rounded-2xl border-2 transition-all flex items-center gap-4 bg-green-50 border-green-500 ring-1 ring-green-500";

                    letter.className = "w-10 h-10 rounded-xl flex items-center justify-center font-black shrink-0 transition-colors bg-green-500 text-white shadow-md";

                    text.classList.add("text-green-800", "font-bold");

                    checkIcon.classList.remove('hidden');

                } else if (buttonIdx === idx) {

                    btn.className = "w-full text-left p-5 rounded-2xl border-2 transition-all flex items-center gap-4 bg-red-50 border-red-300 opacity-80";

                    letter.className = "w-10 h-10 rounded-xl flex items-center justify-center font-black shrink-0 transition-colors bg-red-200 text-red-600";

                    text.classList.add("text-red-700", "line-through", "decoration-red-300");

                    crossIcon.classList.remove('hidden');

                } else {

                    btn.className = "w-full text-left p-5 rounded-2xl border-2 transition-all flex items-center gap-4 opacity-40 border-slate-100 bg-slate-50";

                }

            });

 

            document.getElementById('explanation-text').textContent = q.exp;

            document.getElementById('explanation-box').classList.remove('hidden');

            document.getElementById('next-btn-text').textContent = (currentIndex + 1 === currentQuestions.length) ? "完成測驗查看成績" : "下一題";

        };

 

        window.nextQuestion = async function() {

            if (currentIndex + 1 < currentQuestions.length) {

                currentIndex++;

                showExplanation = false;

                selectedOption = null;

                renderQuestion();

            } else {

                finishQuiz();

            }

        };

 

        async function finishQuiz() {

            const finalScore = Math.round((userScore / currentQuestions.length) * 100);

            document.getElementById('result-name').textContent = userName;

            document.getElementById('final-percentage').textContent = finalScore;

            document.getElementById('final-correct').textContent = userScore;

            document.getElementById('final-total').textContent = `答對題數 / ${currentQuestions.length}`;

           

            switchView('result-view');

 

            // Save to Firebase

            if (currentUser && userName.trim() !== '') {

                try {

                    await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'leaderboard'), {

                        name: userName.trim(),

                        score: finalScore,

                        correct: userScore,

                        total: currentQuestions.length,

                        mode: quizMode === 0 ? "隨機挑戰 (20題)" : `第 ${quizMode} 章 (50題)`,

                        timestamp: serverTimestamp(),

                        uid: currentUser.uid

                    });

                } catch (e) {

                    console.error("Save score error:", e);

                }

            }

        }

 

        window.renderLeaderboard = function() {

            const tbody = document.getElementById('leaderboard-body');

            tbody.innerHTML = '';

           

            if (leaderboardData.length === 0) {

                tbody.innerHTML = `<tr><td colspan="4" class="p-16 text-center text-slate-400 font-medium">尚無測驗紀錄，搶當第一名！</td></tr>`;

                return;

            }

 

            leaderboardData.forEach((entry, idx) => {

                let rankVisual = `<span class="font-bold text-slate-400 px-2">${idx + 1}</span>`;

                if (idx === 0) rankVisual = `<span class="text-2xl">🥇</span>`;

                if (idx === 1) rankVisual = `<span class="text-2xl">🥈</span>`;

                if (idx === 2) rankVisual = `<span class="text-2xl">🥉</span>`;

 

                tbody.innerHTML += `

                    <tr class="border-b border-slate-50 hover:bg-blue-50/50 transition-colors">

                        <td class="p-4">${rankVisual}</td>

                        <td class="p-4 font-bold text-slate-700">${entry.name}</td>

                        <td class="p-4 text-slate-400 text-sm font-medium hidden md:table-cell">${entry.mode}</td>

                        <td class="p-4 text-right font-black text-blue-600 text-lg">

                            ${entry.score}

                            <span class="text-xs text-slate-400 block font-bold">${entry.correct}/${entry.total} 題</span>

                        </td>

                    </tr>

                `;

            });

        };
