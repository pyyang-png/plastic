                {leaderboardData.map((entry, idx) => (

                  <li 

                    key={entry.id} 

                    className={`flex items-center p-4 md:px-8 hover:bg-gray-50 transition-colors ${

                      user && entry.uid === user.uid ? 'bg-gray-50/50' : ''

                    }`}

                  >

                    <div className={`w-6 h-6 flex items-center justify-center text-xs font-medium mr-4 flex-shrink-0 ${

                      idx === 0 ? 'text-gray-900 font-bold' :

                      idx === 1 ? 'text-gray-700' :

                      idx === 2 ? 'text-gray-600' :

                      'text-gray-400'

                    }`}>

                      {idx + 1}.

                    </div>

                    

                    <div className="flex-1 min-w-0 pr-4">

                      <div className="font-medium text-sm md:text-base text-gray-900 truncate flex items-center gap-2" title={entry.name}>

                        {entry.name}

                        {user && entry.uid === user.uid && <span className="text-[10px] border border-gray-300 text-gray-500 px-1.5 py-0.5 rounded leading-none">YOU</span>}

                      </div>

                      <div className="text-[11px] md:text-xs text-gray-500 mt-0.5 flex gap-2">

                        <span>{entry.chapter === 'all' ? '全部' : `Ch.${entry.chapter}`}</span>

                        {entry.timestamp && <span className="text-gray-300">|</span>}

                        {entry.timestamp && <span>{new Date(entry.timestamp.toMillis()).toLocaleDateString()}</span>}

                      </div>

                    </div>

                    

                    <div className="text-right flex-shrink-0">

                      <div className="font-semibold text-lg text-gray-900">

                        {entry.score}

                        <span className="text-xs font-normal text-gray-400 ml-0.5">/{entry.total}</span>

                      </div>

                    </div>

                  </li>

                ))}

              </ul>

            )}

          </div>

        </div>

      </div>

    );

  }


  // --- 視圖: 測驗進行中 (簡約風格 + RWD) ---

  const currentQ = quizQuestions[currentQIndex];

  const progress = ((currentQIndex) / quizQuestions.length) * 100;

  const isCorrect = selectedOption?.startsWith(currentQ.ans);


  return (

    <div className="min-h-screen bg-white flex flex-col items-center p-0 md:py-8 md:px-4 font-sans text-gray-900 h-screen overflow-hidden">

      

      {/* 測驗區主容器 */}

      <div className="w-full max-w-3xl flex flex-col h-full md:border md:border-gray-200 md:rounded-lg md:shadow-sm bg-white">

        

        {/* 頂部導航與進度 */}

        <div className="px-4 py-4 md:px-8 md:pt-8 md:pb-6 border-b border-gray-100 flex-shrink-0">

          <div className="flex justify-between items-center mb-4">

            <div className="flex items-center gap-3">

              <span className="text-xs font-semibold tracking-widest text-gray-500 uppercase">

                Chapter {currentQ.chapter}

              </span>

              <button 

                onClick={finishQuiz}

                className="text-[11px] font-medium text-gray-500 border border-gray-200 hover:border-gray-900 hover:text-gray-900 px-2 py-1 rounded transition-colors"

              >

                停止作答

              </button>

            </div>

            <div className="text-xs font-medium text-gray-400 tracking-widest">

              <span className="text-gray-900">{currentQIndex + 1}</span> / {quizQuestions.length}

            </div>

          </div>

          <div className="w-full bg-gray-100 h-1 rounded-full overflow-hidden">

            <div 

              className="bg-gray-900 h-1 rounded-full transition-all duration-300 ease-out"

              style={{ width: `${progress}%` }}

            ></div>

          </div>

        </div>


        {/* 題目與選項區塊 (可滾動) */}

        <div 

          ref={containerRef}

          className="flex-1 overflow-y-auto px-4 py-6 md:p-8"

        >

          <h2 className="text-lg md:text-xl font-medium text-gray-900 mb-8 leading-relaxed">

            {currentQIndex + 1}. {currentQ.q}

          </h2>


          <div className="space-y-3">

            {currentQ.opts.map((opt, idx) => {

              const isSelected = selectedOption === opt;

              const isThisTheCorrectAns = opt.startsWith(currentQ.ans);

              

              let btnClass = "w-full text-left p-4 md:p-5 rounded-md border transition-all duration-200 flex items-start gap-4 ";

              let iconClass = "mt-0.5 flex-shrink-0 ";

              let textClass = "text-sm md:text-base leading-relaxed ";

              

              if (!isAnswered) {

                btnClass += "border-gray-200 hover:border-gray-400 bg-white cursor-pointer";

                textClass += "text-gray-700";

              } else {

                btnClass += "cursor-default ";

                if (isThisTheCorrectAns) {

                  btnClass += "border-gray-900 bg-gray-50"; // 簡約的正確標示 (深灰邊框+極淺灰底)

                  textClass += "text-gray-900 font-medium";

                } else if (isSelected && !isThisTheCorrectAns) {

                  btnClass += "border-gray-300 bg-white opacity-60"; // 簡約的錯誤標示 (淡化)

                  textClass += "text-gray-500 line-through"; // 錯誤選項加刪除線

                } else {

                  btnClass += "border-gray-100 bg-white opacity-40";

                  textClass += "text-gray-400";

                }

              }


              return (

                <button 

                  key={idx} 

                  onClick={() => handleOptionClick(opt)}

                  disabled={isAnswered}

                  className={btnClass}

                >

                  <div className={iconClass}>

                    {isAnswered && isThisTheCorrectAns && <Check className="w-5 h-5 text-gray-900" />}

                    {isAnswered && isSelected && !isThisTheCorrectAns && <X className="w-5 h-5 text-gray-400" />}

                    {(!isAnswered || (!isSelected && !isThisTheCorrectAns)) && <div className="w-5 h-5 rounded-full border border-gray-300"></div>}

                  </div>

                  <span className={textClass}>

                    {opt}

                  </span>

                </button>

              );

            })}

          </div>


          {/* 簡約解析區 (作答後顯示) */}

          {isAnswered && (

            <div className="mt-10 animate-fade-in">

              <div className="border-l-2 border-gray-900 pl-4 md:pl-6 py-1">

                <div className="mb-4">

                  <span className="text-xs font-semibold text-gray-400 uppercase tracking-widest block mb-1">正確答案</span>

                  <p className="text-gray-900 font-medium text-sm md:text-base">

                    {currentQ.opts.find(opt => opt.startsWith(currentQ.ans))}

                  </p>

                </div>

                <div>

                  <span className="text-xs font-semibold text-gray-400 uppercase tracking-widest block mb-1">解析說明</span>

                  <p className="text-gray-600 text-sm md:text-base leading-relaxed">

                    {currentQ.exp}

                  </p>

                </div>

              </div>

            </div>

          )}

        </div>


        {/* 底部導航 */}

        {isAnswered && (

          <div className="px-4 py-4 md:px-8 md:py-6 border-t border-gray-100 bg-white flex justify-end flex-shrink-0">

            <button 

              onClick={nextQuestion}

              className="w-full md:w-auto px-6 py-3 bg-gray-900 text-white rounded-md font-medium text-sm hover:bg-gray-800 transition-colors flex items-center justify-center gap-2"

            >

              {currentQIndex < quizQuestions.length - 1 ? (

                <>下一題 <ChevronRight className="w-4 h-4" /></>

              ) : (

                <>完成測驗</>

              )}

            </button>

          </div>

        )}

      </div>

      

      <style dangerouslySetInnerHTML={{__html: `

        .animate-fade-in {

          animation: fadeIn 0.4s ease-out forwards;

        }

        @keyframes fadeIn {

          from { opacity: 0; transform: translateY(5px); }

          to { opacity: 1; transform: translateY(0); }

        }

      `}} />

    </div>

  );
